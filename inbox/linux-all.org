*** cpu info
#+BEGIN_SRC sh
  #查询系统具有多少个逻辑核：
  cat /proc/cpuinfo | grep "processor" | wc -l

  #查询系统CPU的物理核数：
  cat /proc/cpuinfo | grep "cpu cores" | uniq

  #查询系统CPU是否启用超线程：
  cat /proc/cpuinfo | grep -e "cpu cores"  -e "siblings" | sort | uniq
#+END_SRC
